import torch
import transformers
from transformers import ElectraTokenizer, ElectraForSequenceClassification, Trainer, TrainingArguments
from datasets import Dataset
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, precision_recall_fscore_support

#  Veri setini ykleme
df = pd.read_csv("c:/malicious_phish/malicious_phish.csv")
labels_dict = {"benign": 0, "defacement": 1, "phishing": 2, "malware": 3}
df["label"] = df["type"].map(labels_dict)

#  Veriyi eitim ve test olarak ayrma
train_texts, test_texts, train_labels, test_labels = train_test_split(
    df["url"].tolist(), df["label"].tolist(), test_size=0.2, random_state=42
)

#  ELECTRA Tokenizer
tokenizer = ElectraTokenizer.from_pretrained("google/electra-small-discriminator")

#  Veriyi tokenize etme
train_encodings = tokenizer(train_texts, truncation=True, padding=True, max_length=150)
test_encodings = tokenizer(test_texts, truncation=True, padding=True, max_length=150)

#  PyTorch Dataset snf
class URLDataset(torch.utils.data.Dataset):
    def __init__(self, encodings, labels):
        self.encodings = encodings
        self.labels = labels

    def __len__(self):
        return len(self.labels)

    def __getitem__(self, idx):
        item = {key: torch.tensor(val[idx]) for key, val in self.encodings.items()}
        item["labels"] = torch.tensor(self.labels[idx])
        return item

train_dataset = URLDataset(train_encodings, train_labels)
test_dataset = URLDataset(test_encodings, test_labels)

#  Modeli ykleme
model = ElectraForSequenceClassification.from_pretrained("google/electra-small-discriminator", num_labels=4)

#  Eitim argmanlar
training_args = TrainingArguments(
    output_dir="./results",
    evaluation_strategy="epoch",
    save_strategy="epoch",
    num_train_epochs=5,
    per_device_train_batch_size=16,
    per_device_eval_batch_size=16,
    warmup_steps=500,
    weight_decay=0.01,
    logging_dir="./logs",
    logging_steps=10,
)

#  Deerlendirme fonksiyonu
def compute_metrics(pred):
    labels = pred.label_ids
    preds = pred.predictions.argmax(-1)
    acc = accuracy_score(labels, preds)
    precision, recall, f1, _ = precision_recall_fscore_support(labels, preds, average="weighted")
    return {"accuracy": acc, "f1": f1, "precision": precision, "recall": recall}

#  Trainer objesi
trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=train_dataset,
    eval_dataset=test_dataset,
    compute_metrics=compute_metrics,
)

#  Modeli eitme
trainer.train()
